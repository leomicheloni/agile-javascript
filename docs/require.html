Permite manejar las dependecias
Carga dinámica y caché
Principales funciones:
define([], function(){
	var A = {
		hacer: function(){
			console.log("hola");
		}
	};
	return A;
});

require(["a"], function(A){
	A.hacer();
});
HTML para empezar
<html>
<script type="text/javascript" src="require.js" data-main="main.js"></script>
</html>
Definir paths
require.config({
	baseUrl: "js",
	paths:{
		"jquery": "lib/jquery",
		"underscore": "lib/underscore"
		"backbone": "lib/backbone"
	},
	shim:{
		"backbone": {
			"exports": "Backbone",
			"dependencies":{
				"underscore"
			}
		}
	}
});

Los paths permites resumir las rutas de las dependecias o cambiar su nombre
También podemos cambiarlas cuando hacemos test para inyectar mocks sin tocar nuestro código
Después de todo Requirejs no es más que un IoC
Shim me permite indicar cómo tratar módulos que no soportar Requirejs
